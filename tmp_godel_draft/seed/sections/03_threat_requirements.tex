\section{What a same-domain G\"odel threat would require}
\label{sec:target}

\subsection{Target regime}

\begin{definition}[Fixed domain and same-domain threat]
A \emph{fixed domain} is a regime in which the admissibility boundary, standing-bearing reuse conditions, lawful same-scope transitions, and closure-sensitive continuation structure are already fixed. A \emph{same-domain threat} is any distinction, construction, or architecture that, without explicit scope change, changes what may be admitted, rejected, reused, committed, or licensed as same-domain continuation on that fixed domain.
\end{definition}

\begin{definition}[Inferential life]
The \emph{inferential life} of the fixed domain is the totality of all same-domain admissions, rejections, standing-bearing reuses, commitments, and legal same-domain continuations supported by that domain.
\end{definition}

\begin{definition}[Code layer, readback, and governance relevance]
Let $A$ be a class of syntax-bearing acts on the fixed domain and let $C$ be a class of code objects. A \emph{same-domain code layer} is a coding map $\ulcorner\cdot\urcorner:A\to C$. For a predicate $P\subseteq C$, its \emph{readback} to the original acts is
\[
\widehat{P}(a) :\Longleftrightarrow P(\ulcorner a\urcorner).
\]
The predicate $P$ is \emph{governance-relevant} if its readback changes inferential life on the fixed domain.
\end{definition}

\begin{proposition}[Recorded coherent-reasoning kernel consequences]
\label{prop:kernel-record}
The present paper works over a fixed kernel in which the following already hold on the same fixed domain:
\begin{enumerate}[label=(K\arabic*), leftmargin=2.8em]
    \item coherence of a legal same-domain trajectory is pointwise standing-preservation along that trajectory;
    \item no deferred coherence: an incoherent prefix remains incoherent under every admissible continuation;
    \item endpoint certification: terminal standing fully certifies coherence of a legal same-domain trajectory;
    \item tensor determination and skin neutrality: all coherence-relevant distinctions factor through standing quotient and tensor content, while skin carries no inferential authority; and
    \item standing conservation and irrecoverability: there is no same-scope creation, amplification, repair, or recovery of standing inside the regime.
\end{enumerate}
\end{proposition}

\begin{remark}[Why \cref{prop:kernel-record} matters]
These are the exact coherent-reasoning results that defeat the hardest hostile attack. The attack says that a same-domain G\"odel threat might hide in intermediate path structure without surfacing in standing classification or terminal trajectory behavior. \Cref{prop:kernel-record} rules that out: pointwise standing-preservation, no-deferred-coherence, endpoint certification, and tensor determination leave no independent hidden path-locus with governance force.
\end{remark}

\begin{lemma}[Inferential-life exhaustion]
\label{lem:inferential-life-exhaustion}
Inferential life on the fixed domain is exhausted by two and only two loci:
\begin{enumerate}[label=(\roman*), leftmargin=2.5em]
    \item the standing classification of evaluated acts; and
    \item the coherent legal trajectory structure of same-domain histories, equivalently their pointwise standing-preservation and terminal standing behavior.
\end{enumerate}
If two same-domain descriptions agree on those loci, then they agree on what may be admitted, rejected, reused, or licensed as same-domain continuation.
\end{lemma}

\begin{proof}
By \cref{prop:kernel-record}, any same-domain act that matters to admission, rejection, or reuse already appears in the standing classification of evaluated acts. Likewise, any same-domain lineage that matters to closure-sensitive continuation is governed by the coherent-reasoning layer: coherence is pointwise standing-preservation on legal same-domain trajectories, no deferred coherence exists, and terminal standing certifies coherence. Therefore any governance-relevant difference in inferential life must appear either as a difference in standing classification or as a difference in coherent legal trajectory structure. Since skin carries no inferential authority and no second same-domain standing-bearing equivalence survives, there is no third governance-bearing locus left inside the regime.
\end{proof}

\begin{lemma}[Closure-relevance transmission]
\label{lem:closure-transmission}
If a same-domain distinction changes inferential life, then it changes either the standing classification of some evaluated act or the coherent legal trajectory / terminal-standing behavior of some legal same-domain trajectory.
\end{lemma}

\begin{proof}
This is the contrapositive reading of \cref{lem:inferential-life-exhaustion}. If a distinction changed neither standing classification nor coherent legal trajectory structure, then it would change nothing about what may be admitted, rejected, reused, or legally continued on the fixed domain. It would therefore be closure-inert.
\end{proof}

\begin{theorem}[Necessary package for a same-domain G\"odel threat]
\label{thm:godel-prerequisites}
If a G\"odel-style phenomenon is to threaten same-domain closure from within the original domain, it must realize all of the following together:
\begin{enumerate}[label=(\roman*), leftmargin=2.6em]
    \item a faithful same-domain coding layer;
    \item a governance-relevant code predicate or readback condition;
    \item a negative fixed-point or diagonal route capable of producing a same-domain closure-sensitive distinction; and
    \item survival of that distinction into standing classification or coherent legal trajectory structure on the original domain.
\end{enumerate}
\end{theorem}

\begin{proof}
A same-domain incompleteness objection matters only if it changes inferential life. By \cref{lem:closure-transmission}, that requires survival into standing classification or coherent legal trajectory structure. But no such change can arise unless there is a code layer, a governance-relevant code predicate, a same-domain readback, and a negative fixed-point route capable of feeding the code predicate back onto original-domain content. Remove any one of these and the architecture is either external metamathematics, mere bookkeeping, or a non-threatening definitional restatement.
\end{proof}
