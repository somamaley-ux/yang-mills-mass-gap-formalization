\paragraph{C3-U inputs.}
The no-subsequence convergence (C3-U) is obtained by combining:
\begin{itemize}
\item C3.1: uniform bounds for flowed observables (precompactness), provided by the polynomial moment/cumulant controls;
\item C3.2(b): injectivity/monotonicity of the renormalization map (Theorem~\ref{thm:rho_inj});
\item C3.3: stable manifold / slaving of the irrelevant sector to the AF coupling trajectory (Theorem~\ref{thm:stable});
\item C3.4: bulk observable Lipschitz dependence on RG data at the observation scale (Theorem~\ref{thm:C3_4}).
\end{itemize}

\begin{quote}
\emph{For a fixed renormalization condition $u(\mu)=u_\mu$ in the AF ball, the chart-truncated RG outputs converge (no subsequence extraction) to a unique continuum state on the flowed observable algebra.  By Theorem~\ref{thm:Wilson_patch_transport}, the unpatched Wilson lattice expectations converge to the same limit.}
\end{quote}

\paragraph{Role of patching.}
All RG estimates are proved using the chart-truncated one-step integration operator (Definitions~\ref{def:trunc_measure} and~\ref{def:RG_good}) but patching is \emph{analytic only}.
Section~\ref{sec:patching} proves the exact Wilson recursion theorem (Theorem~\ref{thm:wilson_exact_recursion}) and the Wilson-to-chart-truncated transport step (Theorem~\ref{thm:Wilson_patch_transport}); together they identify the RG limit with the physical Wilson limit.

Let $\mathcal A_{\mathrm{flow}}$ denote the $\ast$-algebra generated by finite products of gauge-invariant flowed observables
$\mathcal O_{t}(f)$ with $t\in(0,t_\ast]$ and $f\in C_c^\infty(\mathbb R^4)$, together with small flowed loops (if included in the generator set).

For each lattice spacing $a$, let $\omega_a$ be the \emph{unpatched Wilson} lattice Yang--Mills expectation, viewed as a functional on the lattice realization of $\mathcal A_{\mathrm{flow}}$.
By Theorem~\ref{thm:RP_Wilson}, $\omega_a$ is reflection positive on the standard \emph{unflowed} positive-time algebra $\mathcal A_{+}$ of bounded cylinder functions supported in $\{x_0>0\}$ (hence in particular on the gauge-invariant subalgebra).
\emph{We do not assert reflection positivity on the full flowed algebra $\mathcal A_{\mathrm{flow}}$}: gradient flow is parabolic in the Euclidean time direction and mixes across a reflection hyperplane, so Wilson reflection positivity does not transfer to flowed observables by a support argument.

Let $\omega_a^{\mathrm{RG}}$ denote the finite-$a$ functional produced by the patched/chart-truncated dyadic RG recursion induced by the chart-truncated RG map $\mathcal R_j^{\mathrm{good}}$ (Definition~\ref{def:RG_good}), evaluated on observables in $\mathcal A_{\mathrm{flow}}$.
No positivity property is assumed or required for $\omega_a^{\mathrm{RG}}$; it is an analytic surrogate used to control the continuum limit.

We isolate the exact hypotheses required for the closure theorem and point to the modules that provide them.

\begin{proposition}[C3.1 uniform bounds / precompactness (proved using Theorem~\ref{thm:poly_cumulant})]\label{ass:precompact}
For every finite set $\mathcal F\subset\mathcal A_{\mathrm{flow}}$,
\[
\sup_{a\le a_0}\ \max_{F\in\mathcal F}\ |\omega^{\mathrm{RG}}_a(F)|\ <\ \infty.
\]
\end{proposition}

\begin{proposition}[C3.2(b) injective tuning (proved in Theorem~\ref{thm:rho_inj})]\label{ass:inj}
Fix a reference physical scale $\mu$ (dyadic index $j=0$). For each UV cutoff depth $N(a)$, define the finite-cutoff renormalization map
$\rho_N(u_{-N})=u_0$ along the patched tuned trajectory.
Assume that on a uniform AF interval $[0,u_\circ]$, $\rho_N$ is strictly increasing and hence invertible (Theorem~\ref{thm:rho_inj}).
\end{proposition}

\begin{proposition}[C3.3 stable manifold (proved in Theorem~\ref{thm:stable})]\label{ass:stable}
Given an admissible AF coupling trajectory $(u_j)_{j\le 0}$, there exists a unique UV-stable irrelevant trajectory $(K_j)_{j\le 0}$
solving the patched RG recursion and satisfying $K_j\to 0$ as $j\to -\infty$, and it is universal w.r.t. UV irrelevant initialization (Theorem~\ref{thm:stable}).
\end{proposition}

\begin{proposition}[C3.4 bulk observable Lipschitz (proved in Theorem~\ref{thm:C3_4})]\label{ass:Lip}
For each finite $\mathcal F\subset\mathcal A_{\mathrm{flow}}$, there exists an observation scale $j_{\mathrm{obs}}(\mathcal F)\le 0$ and a map
$\mathcal G_{\mathcal F}:(u_{j_{\mathrm{obs}}},K_{j_{\mathrm{obs}}})\mapsto (\omega(F))_{F\in\mathcal F}$
that is Lipschitz on the AF ball with constant $L_{\mathcal F}$ independent of $a$ (Theorem~\ref{thm:C3_4}).
\end{proposition}

\paragraph{Wilson-identification input.}
Section~\ref{sec:patching} proves the exact one-step Wilson block factorization (Theorem~\ref{thm:wilson_one_step_factorization}), the nested exact Wilson recursion theorem (Theorem~\ref{thm:wilson_exact_recursion}), and the Wilson-to-chart-truncated transport theorem (Theorem~\ref{thm:Wilson_patch_transport}).
These are the only extra ingredients needed to identify the chart-truncated RG continuum state with the physical Wilson state.

\begin{theorem}[C3-U closure]\label{thm:C3U}
Assume Propositions~\ref{ass:precompact}--\ref{ass:Lip}.
Fix a target renormalized coupling value $u(\mu)=u_\mu$ in the AF interval, and for each lattice spacing $a$ tune the UV coupling so that the renormalization condition holds.

Then:
\begin{enumerate}[label=(\alph*)]
\item (\textbf{No-subsequence convergence for the RG outputs}) For every finite set $\mathcal F\subset\mathcal A_{\mathrm{flow}}$,
the joint values $\{\omega^{\mathrm{RG}}_a(F)\}_{F\in\mathcal F}$ converge as $a\to 0$ (without subsequence extraction).

\item (\textbf{Unique physical limit state}) There exists a unique state $\omega$ on $\mathcal A_{\mathrm{flow}}$ such that for all $F\in\mathcal A_{\mathrm{flow}}$,
\[
\omega(F)=\lim_{a\to 0}\ \omega_a(F)=\lim_{a\to 0}\ \omega^{\mathrm{RG}}_a(F).
\]

\item (\textbf{Symmetries}) The limit state $\omega$ is Euclidean invariant and reflection invariant (and inherits the remaining \emph{algebraic} OS identities
that are preserved under pointwise limits of the corresponding finite-$a$ identities).
Reflection positivity is used only for the \emph{unflowed} Wilson/continuum state on $\mathcal A_{+}$ (Section~\ref{sec:patching}); it is not assumed (and not needed) on $\mathcal A_{\mathrm{flow}}$.
\end{enumerate}
\end{theorem}

\begin{proof}
Fix a finite set $\mathcal F$ and choose $j_{\mathrm{obs}}=j_{\mathrm{obs}}(\mathcal F)$ from Proposition~\ref{ass:Lip}.
For each $a$, the tuned UV coupling (Proposition~\ref{ass:inj}) determines a unique AF coupling trajectory $(u^{(a)}_j)_{j\le 0}$.
By Proposition~\ref{ass:stable}, there is a unique compatible irrelevant trajectory $(K^{(a)}_j)$ on the UV-stable manifold.

At the observation scale, we thus obtain RG data $(u^{(a)}_{j_{\mathrm{obs}}},K^{(a)}_{j_{\mathrm{obs}}})$.
As $a\to 0$, the cutoff depth $N(a)\to\infty$ and the stable-manifold universality together with injective tuning yield that the RG data at fixed $j_{\mathrm{obs}}$
converge to a unique limit $(u^\infty_{j_{\mathrm{obs}}},K^\infty_{j_{\mathrm{obs}}})$.

Now apply the Lipschitz map $\mathcal G_{\mathcal F}$:
\[
\max_{F\in\mathcal F}|\omega^{\mathrm{RG}}_a(F)-\omega^{\mathrm{RG}}_{a'}(F)|
\le
L_{\mathcal F}\Big(|u^{(a)}_{j_{\mathrm{obs}}}-u^{(a')}_{j_{\mathrm{obs}}}|+\|K^{(a)}_{j_{\mathrm{obs}}}-K^{(a')}_{j_{\mathrm{obs}}}\|_{j_{\mathrm{obs}}}\Big)\ \xrightarrow[a,a'\to 0]{}\ 0,
\]
so $\{\omega^{\mathrm{RG}}_a(F)\}_{a\downarrow0}$ is Cauchy for each $F\in\mathcal F$ and hence convergent. This proves (a).

For (b), Theorem~\ref{thm:Wilson_patch_transport} implies $\omega_a(F)-\omega_a^{\mathrm{RG}}(F)\to 0$ for each $F$, hence the unpatched Wilson expectations converge with the same limit.

For (c), Euclidean invariance is inherited because it is expressed by finite collections of identities at each $a$
and is stable under pointwise limits.
\end{proof}

\begin{corollary}[Unique continuum Euclidean state on $\mathcal A_{\mathrm{flow}}$]\label{cor:flow_state}
Under the hypotheses of Theorem~\ref{thm:C3U}, the tuned unpatched Wilson expectations $\omega_a$ converge (without subsequences) to a unique Euclidean-invariant state
$\omega$ on $\mathcal A_{\mathrm{flow}}$ for all flow times $t\in(0,t_\ast]$.
\end{corollary}

\begin{proof}
This is immediate from Theorem~\ref{thm:C3U}(b)--(c).
\end{proof}

\begin{corollary}[Tuned dyadic Wilson limit on the bounded positive-time algebra]\label{cor:C3U_Aplus_tuned}
Fix a target value $u_{\mathrm{ren}}\in(0,u_\ast]$ and let
\[
a_n:=\ell_0 2^{-n},\qquad \beta_n:=\beta(a_n;u_{\mathrm{ren}})
\]
be the associated tuned dyadic trajectory.
Then for every finite family $\mathcal F\subset\mathcal A_+$ of bounded cylinder observables supported in $\{x_0>0\}$, the Wilson expectations
\[
\{\omega_{a_n}(F)\}_{F\in\mathcal F}
\]
converge jointly as $n\to\infty$.
In particular, for every $F\in\mathcal A_+$ the limit $\lim_{n\to\infty}\omega_{a_n}(F)$ exists and defines a state on $\mathcal A_+$.
\end{corollary}

\begin{proof}
The same C3-U argument applies to finite families of bounded unflowed cylinder observables in $\mathcal A_+$: the observable-evaluation Lipschitz step and the Wilson-to-chart-truncated transport theorem are formulated for finite observable sets and do not require positive flow time.
Thus the tuned Wilson expectations are Cauchy on every finite family in $\mathcal A_+$, yielding the stated joint convergence and the limiting state.
\end{proof}

\begin{corollary}[Tuned dyadic convergence of reflected matrix elements]\label{cor:C3U_reflected_tuned}
Fix a target value $u_{\mathrm{ren}}\in(0,u_\ast]$ and let
\[
a_n:=\ell_0 2^{-n},\qquad \beta_n:=\beta(a_n;u_{\mathrm{ren}})
\]
be the associated tuned dyadic trajectory.
Define the dense dyadic translation set
\[
\mathbb D_{\mathrm{dyad}}:=\{m\ell_0 2^{-k}:m,k\in\mathbb N_0\}.
\]
Then for every finite family of bounded cylinder observables built from finitely many elements of $\mathcal A_+$ by applying finitely many reflections and finitely many translations $\tau_s$ with $s\in\mathbb D_{\mathrm{dyad}}$, the Wilson expectations converge jointly along the tuned dyadic sequence (for all sufficiently large $n$ so that every listed dyadic shift is compatible with the lattice spacing $a_n$).
In particular, for every bounded $F,G\in\mathcal A_+$ and every $s\in\mathbb D_{\mathrm{dyad}}$, the limit
\[
\lim_{n\to\infty}\omega_{a_n}(\Theta F\,\tau_s G)
\]
exists.
\end{corollary}

\begin{proof}
For a fixed finite list of dyadic shifts $s=r\ell_0 2^{-k}$, every sufficiently fine lattice spacing $a_n=\ell_0 2^{-n}$ with $n\ge k$ makes $s/a_n\in\mathbb N$.
Each reflected/translated observable in the stated family is then a bounded unflowed cylinder observable depending on finitely many lattice variables.
The same C3-U convergence argument used in Corollary~\ref{cor:C3U_Aplus_tuned} applies to any such finite family of bounded cylinder observables, because the observable-evaluation Lipschitz step and the Wilson-to-chart-truncated transport theorem are formulated for arbitrary finite observable sets.
Hence the joint tuned Wilson expectations are Cauchy and therefore converge.
\end{proof}
