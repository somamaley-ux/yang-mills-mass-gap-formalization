We now address C3.4: Lipschitz dependence of a finite family of flowed observables on the RG data at the observation scale.
\medskip
\noindent\textbf{Target (C3.4).}
Fix a finite set of flowed observables $\mathcal F\subset\mathcal A_{\mathrm{flow}}$ supported in a fixed compact region and with flow times in $(0,t_\ast]$.
There exists a dyadic scale $j_{\mathrm{obs}}$ (depending only on the geometric support and the minimal flow time in $\mathcal F$) and a map
\[
\mathcal G_{\mathcal F}:\ (u_{j_{\mathrm{obs}}},K_{j_{\mathrm{obs}}})\ \mapsto\ (\omega^{\mathrm{good}}(F))_{F\in\mathcal F}
\]
such that $\mathcal G_{\mathcal F}$ is \textbf{Lipschitz} on the AF ball, with constant independent of lattice spacing $a$ in the patched construction:
\[
\max_{F\in\mathcal F}\big|\omega^{\mathrm{good}}_{(u,K)}(F)-\omega^{\mathrm{good}}_{(\widetilde u,\widetilde K)}(F)\big|
\ \le\ L_{\mathcal F}\Big(|u-\widetilde u|+\|K-\widetilde K\|_{j_{\mathrm{obs}}}\Big).
\]

\paragraph{Why this closes C3-U.}
After Theorem~\ref{thm:stable} (unique $K$ for each AF $u$-trajectory) and Theorem~\ref{thm:rho_inj} (unique tuning for fixed $u(\mu)$), the RG data at scale $j_{\mathrm{obs}}$ become unique as $a\to 0$.
Lipschitz dependence then forces all cluster points of $\{\omega_a^{\mathrm{good}}\}$ to agree on a dense set of observables, hence the continuum state is unique.

\paragraph{Patched setting.}
All statements are in the patched-chart dyadic RG and compatible with the chart-truncated RG step (Definition~\ref{def:RG_good}) and the de-patching bounds of Section~\ref{sec:patching}.

$\mathcal A_{\mathrm{flow}}$ is the $\ast$-algebra generated by finite products of smeared flowed composites:
\[
\mathcal O_t(f)=\int d^4x\ f(x)\,\mathcal O_t(x),\qquad t\in(0,t_\ast],
\]
where $\mathcal O_t(x)$ is a gauge-invariant local composite built from the flowed curvature $G_{\mu\nu}(t,x)$, covariant derivatives, and (optionally) small flowed loops.

Fix a finite set $\mathcal F$; let $K_{\mathcal F}$ be a compact set containing the union of supports of the smearing functions of all insertions in all $F\in\mathcal F$.
Let $t_{\min}:=\min\{t:\ t \text{ appears in some }F\in\mathcal F\}$.

Let $\mu$ be the reference renormalization scale; we index dyadic scales so that $j=0$ corresponds to length $\mu^{-1}$.
Define the observation scale $j_{\mathrm{obs}}\le 0$ as the smallest integer such that
\[
2^{j_{\mathrm{obs}}}\mu^{-1}\ \le\ c_0 \sqrt{t_{\min}}
\quad\text{and}\quad
K_{\mathcal F}\ \text{is covered by at most }N_{\mathcal F}\text{ many }j_{\mathrm{obs}}\text{-blocks},
\]
where $c_0$ is the flow-localization constant from Theorem~\ref{thm:flow_local_v91} and $N_{\mathcal F}$ depends only on the geometry of $K_{\mathcal F}$.

\paragraph{Meaning.}
At this scale, every flowed insertion in $\mathcal F$ is quasi-local at range $\lesssim 2^{j_{\mathrm{obs}}}\mu^{-1}$,
so $\mathcal F$ may be treated as an insertion activity localized on a bounded family of $j_{\mathrm{obs}}$-polymers.

At each dyadic scale $j$, the patched RG produces:
\begin{itemize}
\item a scalar AF coupling coordinate $u_j\in[0,u_\ast]$;
\item an irrelevant polymer activity $K_j$ in a Banach space $(\mathscr K_j,\|\cdot\|_j)$ with $\|K_j\|_j\le r_\ast$.
\end{itemize}
We work on the AF ball
\[
\mathbb B_j := [0,u_\ast]\times B_{\mathscr K_j}(0,r_\ast).
\]

The key technical point is that \emph{a general flowed observable} $F\in\mathcal A_{\mathrm{flow}}$ can be represented at scale $j_{\mathrm{obs}}$
as an \emph{insertion activity} $J^{F}_{j_{\mathrm{obs}}}$ supported near $K_{\mathcal F}$, with norm bounded uniformly in $a$ under the patched chart.
This is the same insertion-norm technology used earlier for the small-flow-time remainder analysis, but now without taking a $t\downarrow 0$ limit.

For fixed $j$, define the insertion norm
\begin{equation}\label{eq:ins_norm_simple}
\|J\|_{\mathrm{ins};j}
:=\sup_{X\in\mathscr P_j}\ \Gamma_j(X)^{-1}\ \|J(X)\|_j,
\end{equation}
where $\mathscr P_j$ are $j$-polymers and $\Gamma_j(X)=A^{|X|_j}$ is the polymer weight.

\begin{lemma}[Flow localization $\Rightarrow$ bounded insertion norm for $F$]\label{lem:F_to_J}
For each $F\in\mathcal F$ there exists a $j_{\mathrm{obs}}$-scale insertion activity $J^F_{j_{\mathrm{obs}}}$
supported on polymers intersecting the $j_{\mathrm{obs}}$-neighborhood of $K_{\mathcal F}$ such that:
\begin{enumerate}[label=(\alph*)]
\item $F$ and $J^F_{j_{\mathrm{obs}}}$ define the same observable in the physical Wilson state (any difference produced by the good-chart truncation is an irrelevant $O(u^N)$ term controlled by the de-patching stability lemmas);
\item $\|J^F_{j_{\mathrm{obs}}}\|_{\mathrm{ins};j_{\mathrm{obs}}}\le C_{F,0}$ with $C_{F,0}$ independent of $a$.
\end{enumerate}
\end{lemma}

\begin{proof}
Fix $F\in\mathcal F$.
By construction of the probe family $\mathcal F$, each $F$ is a bounded functional of the flowed field at fixed positive flow time,
with dependence localized near the compact set $K_{\mathcal F}$.
By the flow quasi-locality/derivative bounds (Theorem~\ref{thm:flow_local_v91}), for any choice of truncation radius $R\ge 1$
there exists a truncated representative $F^{(R)}$ depending only on the underlying variables in the $j_{\mathrm{obs}}$-neighborhood
of $K_{\mathcal F}$ such that
\[
\|F-F^{(R)}\|_{L^\infty}\le C\,t^{-5}e^{-cR^2},
\]
with constants uniform in the lattice spacing $a$ on the patched-chart regime.
When passing between the unpatched Wilson expectation and the patched (good-event) expectation, the difference is an irrelevant
$O(u^N)$ term controlled by the de-patching/truncation bounds (Lemma~\ref{lem:patch_error} combined with Lemma~\ref{lem:local_tail}).
Thus, for the purposes of the insertion encoding at the observation scale, we may replace $F$ by a representative supported in a bounded
$j_{\mathrm{obs}}$-neighborhood of $K_{\mathcal F}$.

Let $\mathscr X_F$ be the finite family of $j_{\mathrm{obs}}$-polymers intersecting this neighborhood.
Fix a deterministic choice rule $X_\ast(f)\in\mathscr X_F$ that contains the (bounded) effective support of the smeared observable $F(f)$.
Define the insertion activity at scale $j_{\mathrm{obs}}$ by
\[
J^{F}_{j_{\mathrm{obs}}}(X;f):=\begin{cases}
F(f),& X=X_\ast(f),\\
0,& \text{otherwise.}
\end{cases}
\]
Then $F$ and $J^{F}_{j_{\mathrm{obs}}}$ define the same observable (pointwise on configurations, hence in the Wilson state), proving (a).

For (b), the activity is supported on the fixed finite family $\mathscr X_F$.
By the seminorm axioms on the patched-chart regime (Proposition~\ref{ass:bulk_axioms}) and the uniform derivative bounds from
Theorem~\ref{thm:flow_local_v91}, there is a constant $C_{F,0}$ (depending only on $F$ and the fixed parameters of the construction)
with $\|F(f)\|_{j_{\mathrm{obs}}}\le C_{F,0}\,\Gamma_{j_{\mathrm{obs}}}(X_\ast(f))$ for all admissible $f$.
Therefore
\[
\|J^{F}_{j_{\mathrm{obs}}}\|_{\mathrm{ins};j_{\mathrm{obs}}}
\le \sup_{X\in\mathscr X_F}\Gamma_{j_{\mathrm{obs}}}(X)^{-1}\,\|J^{F}_{j_{\mathrm{obs}}}(X)\|_{j_{\mathrm{obs}}}
\le C_{F,0},
\]
and $C_{F,0}$ is independent of $a$.
\end{proof}

Fix a scale $j$ and a bulk dataset $(u,K)\in\mathbb B_j$.
Let $Z_j^{(u,K)}(\varphi)=e^{-V_j(u;\varphi)}\big(1+K(\varphi)\big)$ be the scale-$j$ effective density.
The patched one-step insertion integration operator is
\[
\mathcal E^{\mathrm{good}}_{j+1}(u,K):\ J_j\ \mapsto\ \widetilde J_{j+1},
\qquad
\widetilde J_{j+1}(\varphi)
=
\frac{\mathbb E_{j+1}^{\mathrm{good}}\!\big[J_j(\varphi+\zeta)Z_j^{(u,K)}(\varphi+\zeta)\big]}
{\mathbb E_{j+1}^{\mathrm{good}}\!\big[Z_j^{(u,K)}(\varphi+\zeta)\big]},
\]
followed by standard polymer re-expansion.
Proposition~\ref{prop:Eins_bound} proves the uniform boundedness:
\begin{equation}\label{eq:Eins_bound}
\|\mathcal E^{\mathrm{good}}_{j+1}(u,K)J\|_{\mathrm{ins};j+1}\le C_{E,\mathrm{ins}}\ \|J\|_{\mathrm{ins};j},
\end{equation}
with explicit $C_{E,\mathrm{ins}}$ uniform in $a$.

We now prove the one-step parameter Lipschitz bound that upgrades the boundedness into C3.4.

\begin{proposition}[Local potential Lipschitz in $u$ on the AF ball (proved in Theorem~\ref{thm:RG_analytic_bounds})]\label{ass:V_Lip}
There exist constants $C_{V,0},C_{V,1},c_{V,\mathrm{den}}>0$ such that for all scales $j$ and all $u,\widetilde u\in[0,u_\ast]$:
\begin{enumerate}[label=(V\arabic*)]
\item $\|e^{-V_j(u)}\|_j\le C_{V,0}$;
\item $\|e^{-V_j(u)}-e^{-V_j(\widetilde u)}\|_j\le C_{V,1}|u-\widetilde u|$;
\item $\|\big(\mathbb E_{j+1}^{\mathrm{good}}[e^{-V_j(u)}]\big)^{-1}\|_{j+1}\le c_{V,\mathrm{den}}^{-1}$.
\end{enumerate}
All constants are uniform in $j$ and $a$ under the patched construction.
\end{proposition}

\begin{lemma}[One-step parameter Lipschitz bound for $\mathcal E^{\mathrm{good}}_{j+1}$]\label{lem:Eins_paramLip}
Assume the constants $C_E$ (one-step expectation bound) and $C_{\mathrm{prod}}$ (product bound),
assume $\|K\|_j,\|\widetilde K\|_j\le r_\ast$ and Proposition~\ref{ass:V_Lip}.
Then there exists an explicit constant $C_{E,\mathrm{par}}$ such that for any insertion activity $J$,
\begin{equation}\label{eq:Eins_paramLip}
\|\mathcal E^{\mathrm{good}}_{j+1}(u,K)J-\mathcal E^{\mathrm{good}}_{j+1}(\widetilde u,\widetilde K)J\|_{\mathrm{ins};j+1}
\le
C_{E,\mathrm{par}}\ \|J\|_{\mathrm{ins};j}\ \Big(|u-\widetilde u|+\|K-\widetilde K\|_j\Big),
\end{equation}
with
\begin{equation}\label{eq:C_Epar}
C_{E,\mathrm{par}}
=
C_{\mathrm{mark}}\ C_E\ c_{V,\mathrm{den}}^{-2}\ C_{\mathrm{prod}}^2\Big(
C_{V,1}(1+r_\ast)+C_{V,0}\Big),
\end{equation}
where $C_{\mathrm{mark}}$ is the marked-polymer re-expansion counting constant.
All constants are uniform in $j$ and $a$ under the patched construction.
\end{lemma}

\begin{proof}
Write the pre-reexpansion object as a ratio $N/D$ with
\[
N(u,K):=\mathbb E_{j+1}^{\mathrm{good}}[J e^{-V(u)}(1+K)],
\qquad
D(u,K):=\mathbb E_{j+1}^{\mathrm{good}}[e^{-V(u)}(1+K)].
\]
Then
\[
\frac{N}{D}-\frac{\widetilde N}{\widetilde D}
=
\frac{N-\widetilde N}{D}
+\widetilde N\Big(\frac{1}{D}-\frac{1}{\widetilde D}\Big).
\]
Use $|1/D|\le c_{V,\mathrm{den}}^{-1}(1-C_E r_\ast)^{-1}$; under RG-ball smallness we absorb $(1-C_E r_\ast)^{-1}$ into $c_{V,\mathrm{den}}^{-1}$.

For $N-\widetilde N$, expand:
\[
e^{-V(u)}(1+K)-e^{-V(\widetilde u)}(1+\widetilde K)
=
\big(e^{-V(u)}-e^{-V(\widetilde u)}\big)(1+K) + e^{-V(\widetilde u)}(K-\widetilde K).
\]
Apply the product bound twice and the expectation bound $\|\mathbb E^{\mathrm{good}}_{j+1}[F]\|_{j+1}\le C_E\|F\|_j$,
plus Proposition~\ref{ass:V_Lip} to bound $\|e^{-V(u)}-e^{-V(\widetilde u)}\|_j$ and $\|e^{-V(\widetilde u)}\|_j$.
Similarly bound $D-\widetilde D$ (with $J$ removed), and then use
\[
\Big\|\frac{1}{D}-\frac{1}{\widetilde D}\Big\|\le \|D^{-1}\|\,\|\widetilde D^{-1}\|\,\|D-\widetilde D\|.
\]
Finally incorporate the marked-polymer re-expansion factor $C_{\mathrm{mark}}$ to pass from a single-polymer bound to the insertion norm.
\end{proof}

We now convert the one-step Lipschitz bound into a Lipschitz theorem for expectations of general flowed observables.

\begin{proposition}[One-step bulk RG Lipschitz constant (proved in Theorem~\ref{thm:RG_analytic_bounds})]\label{ass:bulkLip}
There exists $C_{\mathrm{RG}}\ge 1$ such that for all $j$ and all $(u,K),(\widetilde u,\widetilde K)\in\mathbb B_j$,
\[
|u'-\widetilde u'|+\|K'-\widetilde K'\|_{j+1}
\le
C_{\mathrm{RG}}\Big(|u-\widetilde u|+\|K-\widetilde K\|_{j}\Big),
\]
where $(u',K')=\mathcal R^{\mathrm{good}}_j(u,K)$ and $(\widetilde u',\widetilde K')=\mathcal R^{\mathrm{good}}_j(\widetilde u,\widetilde K)$.
\end{proposition}

Let $n:=0-j_{\mathrm{obs}}\in\mathbb N$ be the number of RG steps from the observation scale to the reference scale.
For each dataset $(u_{j_{\mathrm{obs}}},K_{j_{\mathrm{obs}}})$, define the forward bulk RG trajectory to scale $0$:
\[
(u_{j+1},K_{j+1})=\mathcal R^{\mathrm{good}}_j(u_j,K_j),
\qquad j=j_{\mathrm{obs}},\dots,-1.
\]
For an insertion $J_{j_{\mathrm{obs}}}$, define the forward insertion trajectory:
\[
J_{j+1}=\mathcal E^{\mathrm{good}}_{j+1}(u_j,K_j)J_j,
\qquad j=j_{\mathrm{obs}},\dots,-1.
\]
At scale $0$ we define the (patched) expectation of the insertion by a bounded linear evaluation map
\[
\mathrm{Eval}_0:\ \mathscr J_0\to\mathbb C,
\qquad
|\mathrm{Eval}_0(J_0)|\le C_{\mathrm{eval}}\,\|J_0\|_{\mathrm{ins};0}.
\]
(In finite volume, $C_{\mathrm{eval}}$ can be taken $1$ after normalization; we keep $C_{\mathrm{eval}}$ explicit.)

\begin{theorem}[C3.4: bulk observable Lipschitz in RG data]\label{thm:C3_4}
Assume Lemma~\ref{lem:F_to_J}, the insertion bounds \eqref{eq:Eins_bound}, the parameter Lipschitz bound Lemma~\ref{lem:Eins_paramLip},
and the bulk Lipschitz assumption \ref{ass:bulkLip}.
Fix a finite observable set $\mathcal F$ and let $j_{\mathrm{obs}}$ be as above.
Define for each $F\in\mathcal F$ the map
\[
\mathcal G_F(u_{j_{\mathrm{obs}}},K_{j_{\mathrm{obs}}})
:=\mathrm{Eval}_0\big(J^{F}_{0}\big),
\]
where $J^{F}_0$ is obtained by propagating $J^F_{j_{\mathrm{obs}}}$ along the bulk RG trajectory from $(u_{j_{\mathrm{obs}}},K_{j_{\mathrm{obs}}})$.
Then for any two datasets $(u,K),(\widetilde u,\widetilde K)\in\mathbb B_{j_{\mathrm{obs}}}$,
\begin{equation}\label{eq:mainLip}
\max_{F\in\mathcal F}\big|\mathcal G_F(u,K)-\mathcal G_F(\widetilde u,\widetilde K)\big|
\le
L_{\mathcal F}\Big(|u-\widetilde u|+\|K-\widetilde K\|_{j_{\mathrm{obs}}}\Big),
\end{equation}
with explicit constant
\begin{equation}\label{eq:L_F}
L_{\mathcal F}
=
C_{\mathrm{eval}}\ C_{E,\mathrm{par}}\ C_{\mathcal F,0}\ C_{E,\mathrm{ins}}^{\,n-1}\ \sum_{r=0}^{n-1} C_{\mathrm{RG}}^{\,r},
\qquad
C_{\mathcal F,0}:=\max_{F\in\mathcal F}\|J^F_{j_{\mathrm{obs}}}\|_{\mathrm{ins};j_{\mathrm{obs}}}.
\end{equation}
All constants are uniform in lattice spacing $a$ in the patched construction.
\end{theorem}

\begin{proof}
Fix $F$ and define two coupled trajectories (un-tilde and tilde) from the two initial datasets.
Let $\delta_j:=|u_j-\widetilde u_j|+\|K_j-\widetilde K_j\|_j$.
Proposition~\ref{ass:bulkLip} yields $\delta_{j_{\mathrm{obs}}+r}\le C_{\mathrm{RG}}^{\,r}\delta_{j_{\mathrm{obs}}}$.

Let $D_j:=\|J^F_j-\widetilde J^F_j\|_{\mathrm{ins};j}$. Since the initial insertion is the same, $D_{j_{\mathrm{obs}}}=0$.
At one step,
\[
D_{j+1}
\le
\|\mathcal E_{j+1}(u_j,K_j)(J_j-\widetilde J_j)\|_{\mathrm{ins};j+1}
+
\|(\mathcal E_{j+1}(u_j,K_j)-\mathcal E_{j+1}(\widetilde u_j,\widetilde K_j))\widetilde J_j\|_{\mathrm{ins};j+1}.
\]
Bound the first term by $C_{E,\mathrm{ins}}D_j$ using \eqref{eq:Eins_bound} and the second by Lemma~\ref{lem:Eins_paramLip}:
\[
D_{j+1}\le C_{E,\mathrm{ins}}D_j + C_{E,\mathrm{par}}\|\widetilde J_j\|_{\mathrm{ins};j}\,\delta_j.
\]
Also $\|\widetilde J_j\|_{\mathrm{ins};j}\le C_{E,\mathrm{ins}}^{\,j-j_{\mathrm{obs}}} \|J^F_{j_{\mathrm{obs}}}\|_{\mathrm{ins};j_{\mathrm{obs}}}$.
Iterating from $j=j_{\mathrm{obs}}$ to $-1$ gives
\[
D_0 \le C_{E,\mathrm{par}}\,\|J^F_{j_{\mathrm{obs}}}\|_{\mathrm{ins};j_{\mathrm{obs}}}\ C_{E,\mathrm{ins}}^{\,n-1}\sum_{r=0}^{n-1} C_{\mathrm{RG}}^{\,r}\ \delta_{j_{\mathrm{obs}}}.
\]
Finally,
\[
|\mathcal G_F(u,K)-\mathcal G_F(\widetilde u,\widetilde K)|
\le C_{\mathrm{eval}}\,D_0.
\]
Take the maximum over $F\in\mathcal F$.
\end{proof}

\begin{corollary}[Cluster points coincide on flowed observables]\label{cor:cluster}
Assume Theorems~\ref{thm:stable} and~\ref{thm:rho_inj} so that for fixed renormalization condition $u(\mu)=u_\mu$ the RG data
$(u_{j_{\mathrm{obs}}},K_{j_{\mathrm{obs}}})$ converge to a unique limit as $a\to0$.
Then for every finite set $\mathcal F\subset\mathcal A_{\mathrm{flow}}$, the expectations $\omega_a^{\mathrm{good}}(F)$ converge \emph{without subsequences}
for all $F\in\mathcal F$.
\end{corollary}

\paragraph{Next step (C3-U completion).}
Combine Corollary~\ref{cor:cluster} with the precompactness input C3.1 (from uniform moment/cumulant bounds) to conclude that the full family of patched states
$\omega_a^{\mathrm{good}}$ converges, without subsequences, to a \emph{unique} continuum Euclidean-invariant state on $\mathcal A_{\mathrm{flow}}$.
This is the final packaging theorem C3-U (Theorem~\ref{thm:C3U}).
No reflection-positivity statement is made on $\mathcal A_{\mathrm{flow}}$; reflection positivity is recovered later for the unflowed continuum Wilson state on $\mathcal A_+$.
