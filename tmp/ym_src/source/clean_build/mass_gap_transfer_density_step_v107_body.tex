\paragraph{A dense local subalgebra.}
Let $\mathcal A_{+,\mathrm{bnd}}\subset\mathcal A_+$ be the unflowed bounded gauge-invariant local (cylinder) subalgebra supported in $\{x_0>0\}$.
Let $\mathcal A_{+,\mathrm{poly}}\subset\mathcal A_{+,\mathrm{bnd}}$ be any countable $\ast$-subalgebra that generates
$\mathcal A_{+,\mathrm{bnd}}$ under uniform limits on each finite lattice/cylinder (e.g. the $\ast$-algebra generated by Wilson loops
with rational coefficients, using Peter--Weyl on the compact gauge group).

\begin{lemma}[Polynomial density and capwise graph-core approximation]\label{lem:density_poly}
The span of OS-classes $\{[F]:F\in\mathcal A_{+,\mathrm{poly}}\}$ is dense in the OS Hilbert space $\mathcal H$.
Moreover, whenever a finite family of renormalized local fields belongs to a fixed engineering cap, the corresponding sharp-local field-polynomial vectors are realized as Hilbert-norm limits (indeed finite graph-norm limits) of vectors from the explicit common core furnished by Lemma~\ref{lem:finite_cap_common_core}.
\end{lemma}

\begin{proof}
By construction, $\mathcal H$ is the completion of $\mathcal A_+ / \{\|\cdot\|_{\mathrm{OS}}=0\}$.
On bounded observables one has $\|F\|_{\mathrm{OS}}\le \|F\|_{\infty}$.
Uniform density of $\mathcal A_{+,\mathrm{poly}}$ in $\mathcal A_{+,\mathrm{bnd}}$ therefore implies density of their OS-classes.
Finally, bounded cylinder functions generate $\mathcal A_+$ as a $\ast$-algebra for OS reconstruction, so their OS-classes are dense.
The capwise graph-norm statement is exactly the package recorded in Lemma~\ref{lem:QE3_core_topology}(ii).
\end{proof}

\begin{theorem}[Local-state and graph-core formulation of the gap]\label{thm:gap_loc}
In the setting of Theorem~\ref{thm:gap_transfer}, for every $\psi\in\mathcal H$ with $\langle\Omega,\psi\rangle=0$,
\[
\|e^{-tH}\psi\|\le e^{-\mathfrak m_\ast t}\,\|\psi\|\qquad (t\ge 0).
\]
In particular the same bound holds on the dense subspace generated by polynomial local vectors
$\mathrm{span}\{[F]:F\in\mathcal A_{+,\mathrm{poly}}\}$,
and, after fixing any finite engineering cap, on the capwise graph-core closures of the corresponding sharp-local field polynomials described in Lemma~\ref{lem:density_poly}.
\end{theorem}

\begin{proof}
By the spectral theorem and Theorem~\ref{thm:gap_transfer},
the spectral measure of any $\psi\perp\Omega$ is supported in $[\mathfrak m_\ast,\infty)$, hence
\[
\|e^{-tH}\psi\|\le e^{-\mathfrak m_\ast t}\|\psi\|.
\]
The final sentence follows from density of polynomial local vectors and from the Hilbert/graph-core approximation package cited in Lemma~\ref{lem:density_poly}.
\end{proof}
