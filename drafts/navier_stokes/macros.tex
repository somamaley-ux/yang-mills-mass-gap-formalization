% When multiple statement environments share a counter, cleveref will otherwise
% label them all as the first-defined type. We use aliascnt so that \Cref prints
% the correct environment name (Theorem/Lemma/...) while preserving a single
% statement numbering stream.

\newaliascnt{remark}{definition}
\newtheorem{remark}[remark]{Remark}
\aliascntresetthe{remark}
\crefname{remark}{Remark}{Remarks}
\Crefname{remark}{Remark}{Remarks}

\newaliascnt{example}{definition}
\newtheorem{example}[example]{Example}
\aliascntresetthe{example}
\crefname{example}{Example}{Examples}
\Crefname{example}{Example}{Examples}

\newaliascnt{axiom}{definition}
\newtheorem{axiom}[axiom]{Axiom}
\aliascntresetthe{axiom}
\crefname{axiom}{Axiom}{Axioms}
\Crefname{axiom}{Axiom}{Axioms}

\newaliascnt{rulelabel}{definition}
\newtheorem{rulelabel}[rulelabel]{Rule label}
\aliascntresetthe{rulelabel}
\crefname{rulelabel}{Rule label}{Rule labels}
\Crefname{rulelabel}{Rule label}{Rule labels}

\theoremstyle{plain}

\newtheorem{theorem}[theorem]{Theorem}

\crefname{theorem}{Theorem}{Theorems}

\newaliascnt{proposition}{definition}
